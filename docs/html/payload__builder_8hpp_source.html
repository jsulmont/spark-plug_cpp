22template <typename T>
23concept SparkplugSignedInteger =
24 std::same_as<std::remove_cvref_t<T>, int8_t> || std::same_as<std::remove_cvref_t<T>, int16_t> ||
25 std::same_as<std::remove_cvref_t<T>, int32_t> || std::same_as<std::remove_cvref_t<T>, int64_t>;

28template <typename T>
29concept SparkplugUnsignedInteger = std::same_as<std::remove_cvref_t<T>, uint8_t> ||
30 std::same_as<std::remove_cvref_t<T>, uint16_t> ||
31 std::same_as<std::remove_cvref_t<T>, uint32_t> ||
32 std::same_as<std::remove_cvref_t<T>, uint64_t>;

63template <SparkplugMetricType T>
64consteval DataType get_datatype() {
65 using BaseT = std::remove_cvref_t<T>;
66 if constexpr (std::is_same_v<BaseT, int8_t>)
67 return DataType::Int8;
68 else if constexpr (std::is_same_v<BaseT, int16_t>)
69 return DataType::Int16;
70 else if constexpr (std::is_same_v<BaseT, int32_t>)
71 return DataType::Int32;
72 else if constexpr (std::is_same_v<BaseT, int64_t>)
73 return DataType::Int64;
74 else if constexpr (std::is_same_v<BaseT, uint8_t>)
75 return DataType::UInt8;
76 else if constexpr (std::is_same_v<BaseT, uint16_t>)
77 return DataType::UInt16;
78 else if constexpr (std::is_same_v<BaseT, uint32_t>)
79 return DataType::UInt32;
80 else if constexpr (std::is_same_v<BaseT, uint64_t>)
81 return DataType::UInt64;
82 else if constexpr (std::is_same_v<BaseT, float>)
83 return DataType::Float;
84 else if constexpr (std::is_same_v<BaseT, double>)
85 return DataType::Double;
86 else if constexpr (std::is_same_v<BaseT, bool>)
87 return DataType::Boolean;
88 else
89 return DataType::String;
90}

92template <SparkplugMetricType T>
93void set_metric_value(org::eclipse::tahu::protobuf::Payload::Metric* metric, T&& value) {
94 using BaseT = std::remove_cvref_t<T>;
95
96 if constexpr (std::is_same_v<BaseT, int8_t> || std::is_same_v<BaseT, int16_t> ||
97 std::is_same_v<BaseT, int32_t> || std::is_same_v<BaseT, uint8_t> ||
98 std::is_same_v<BaseT, uint16_t> || std::is_same_v<BaseT, uint32_t>) {
99 metric->set_int_value(value);
100 } else if constexpr (std::is_same_v<BaseT, int64_t> || std::is_same_v<BaseT, uint64_t>) {
101 metric->set_long_value(value);
102 } else if constexpr (std::is_same_v<BaseT, float>) {
103 metric->set_float_value(value);
104 } else if constexpr (std::is_same_v<BaseT, double>) {
105 metric->set_double_value(value);
106 } else if constexpr (std::is_same_v<BaseT, bool>) {
107 metric->set_boolean_value(value);
108 } else {
109 // Handle all string-like types
110 metric->set_string_value(std::string(value));
111 }
112}

114template <SparkplugMetricType T>
115void add_metric_to_payload(org::eclipse::tahu::protobuf::Payload& payload, std::string_view name,
116 T&& value, std::optional<uint64_t> alias,
117 std::optional<uint64_t> timestamp_ms) {
118 auto* metric = payload.add_metrics();
119
120 if (!name.empty()) {
121 metric->set_name(std::string(name));
122 }
123 if (alias.has_value()) {
124 metric->set_alias(*alias);
125 }
126
127 metric->set_datatype(std::to_underlying(get_datatype<T>()));
128 set_metric_value(metric, std::forward<T>(value));
129
130 // Use provided timestamp or current time
131 uint64_t ts;
132 if (timestamp_ms.has_value()) {
133 ts = *timestamp_ms;
134 } else {
135 auto now = std::chrono::system_clock::now();
136 ts = std::chrono::duration_cast<std::chrono::milliseconds>(now.time_since_epoch()).count();
137 }
138 metric->set_timestamp(ts);
139}
