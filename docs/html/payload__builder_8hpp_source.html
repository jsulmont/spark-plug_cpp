19template <typename T>
20consteval DataType get_datatype() {
21 using BaseT = std::remove_cvref_t<T>;
22 if constexpr (std::is_same_v<BaseT, int8_t>)
23 return DataType::Int8;
24 else if constexpr (std::is_same_v<BaseT, int16_t>)
25 return DataType::Int16;
26 else if constexpr (std::is_same_v<BaseT, int32_t>)
27 return DataType::Int32;
28 else if constexpr (std::is_same_v<BaseT, int64_t>)
29 return DataType::Int64;
30 else if constexpr (std::is_same_v<BaseT, uint8_t>)
31 return DataType::UInt8;
32 else if constexpr (std::is_same_v<BaseT, uint16_t>)
33 return DataType::UInt16;
34 else if constexpr (std::is_same_v<BaseT, uint32_t>)
35 return DataType::UInt32;
36 else if constexpr (std::is_same_v<BaseT, uint64_t>)
37 return DataType::UInt64;
38 else if constexpr (std::is_same_v<BaseT, float>)
39 return DataType::Float;
40 else if constexpr (std::is_same_v<BaseT, double>)
41 return DataType::Double;
42 else if constexpr (std::is_same_v<BaseT, bool>)
43 return DataType::Boolean;
44 else
45 return DataType::String;
46}

48template <typename T>
49void set_metric_value(org::eclipse::tahu::protobuf::Payload::Metric* metric, T&& value) {
50 using BaseT = std::remove_cvref_t<T>;
51
52 if constexpr (std::is_same_v<BaseT, int8_t> || std::is_same_v<BaseT, int16_t> ||
53 std::is_same_v<BaseT, int32_t> || std::is_same_v<BaseT, uint8_t> ||
54 std::is_same_v<BaseT, uint16_t> || std::is_same_v<BaseT, uint32_t>) {
55 metric->set_int_value(value);
56 } else if constexpr (std::is_same_v<BaseT, int64_t> || std::is_same_v<BaseT, uint64_t>) {
57 metric->set_long_value(value);
58 } else if constexpr (std::is_same_v<BaseT, float>) {
59 metric->set_float_value(value);
60 } else if constexpr (std::is_same_v<BaseT, double>) {
61 metric->set_double_value(value);
62 } else if constexpr (std::is_same_v<BaseT, bool>) {
63 metric->set_boolean_value(value);
64 } else {
65 // Handle all string-like types
66 metric->set_string_value(std::string(value));
67 }
68}

70template <typename T>
71void add_metric_to_payload(org::eclipse::tahu::protobuf::Payload& payload, std::string_view name,
72 T&& value, std::optional<uint64_t> alias,
73 std::optional<uint64_t> timestamp_ms) {
74 auto* metric = payload.add_metrics();
75
76 if (!name.empty()) {
77 metric->set_name(std::string(name));
78 }
79 if (alias.has_value()) {
80 metric->set_alias(*alias);
81 }
82
83 metric->set_datatype(static_cast<uint32_t>(get_datatype<T>()));
84 set_metric_value(metric, std::forward<T>(value));
85
86 // Use provided timestamp or current time
87 uint64_t ts;
88 if (timestamp_ms.has_value()) {
89 ts = *timestamp_ms;
90 } else {
91 auto now = std::chrono::system_clock::now();
92 ts = std::chrono::duration_cast<std::chrono::milliseconds>(now.time_since_epoch()).count();
93 }
94 metric->set_timestamp(ts);
95}
